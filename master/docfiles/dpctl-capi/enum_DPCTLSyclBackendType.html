Redefinition of DPC++-specific Sycl backend types.

#include <dpctl_sycl_enum_types.h>

enum DPCTLSyclBackendType
{
    DPCTL_CUDA            = 1<<16,
    DPCTL_HOST            = 1<<17,
    DPCTL_LEVEL_ZERO      = 1<<18,
    DPCTL_OPENCL          = 1<<19,
    DPCTL_UNKNOWN_BACKEND = 0,
    DPCTL_ALL_BACKENDS    =((1<<5)-1)<<16,
};
