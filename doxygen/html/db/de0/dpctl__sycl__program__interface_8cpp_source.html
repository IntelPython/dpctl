41 __dpctl_give DPCTLSyclProgramRef
42 createOpenCLInterOpProgram (const context &SyclCtx,
43  __dpctl_keep const void *IL,
44  size_t length)
45 {
46  cl_int err;
47  auto CLCtx = SyclCtx.get();
48  auto CLProgram = clCreateProgramWithIL(CLCtx, IL, length, &err);
49  if (err) {
50  // \todo: record the error string and any other information.
51  std::cerr << "OpenCL program could not be created from the SPIR-V "
52  "binary. OpenCL Error " << err << ".\n";
53  return nullptr;
54  }
55  auto SyclDevices = SyclCtx.get_devices();
56 
57  // Get a list of CL Devices from the Sycl devices
58  auto CLDevices = new cl_device_id[SyclDevices.size()];
59  for (auto i = 0ul; i < SyclDevices.size(); ++i)
60  CLDevices[i] = SyclDevices[i].get();
61 
62  // Build the OpenCL interoperability program
63  err = clBuildProgram(CLProgram, (cl_uint)(SyclDevices.size()), CLDevices,
64  nullptr, nullptr, nullptr);
65  // free the CLDevices array
66  delete[] CLDevices;
67 
68  if (err) {
69  // \todo: record the error string and any other information.
70  std::cerr << "OpenCL program could not be built. OpenCL Error "
71  << err << ".\n";
72  return nullptr;
73  }
74 
75  // Create the Sycl program from OpenCL program
76  try {
77  auto SyclProgram = new program(SyclCtx, CLProgram);
78  return wrap(SyclProgram);
79  } catch (invalid_object_error &e) {
80  // \todo record error
81  std::cerr << e.what() << '\n';
82  return nullptr;
83  }
84 }
