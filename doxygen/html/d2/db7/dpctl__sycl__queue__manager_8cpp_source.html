58  static QVec* init_queues (backend BE, info::device_type DTy)
59  {
60  QVec *queues = new QVec();
61  auto Platforms = platform::get_platforms();
62  for (auto &p : Platforms) {
63  if (p.is_host()) continue;
64  auto be = p.get_backend();
65  auto Devices = p.get_devices();
66 
67  if (Devices.size() == 1) {
68  auto d = Devices[0];
69  auto devty = d.get_info<info::device::device_type>();
70  if(devty == DTy && be == BE) {
71  auto Ctx = context(d);
72  queues->emplace_back(Ctx, d);
73  break;
74  }
75  }
76  else {
77  vector_class<device> SelectedDevices;
78  for(auto &d : Devices) {
79  auto devty = d.get_info<info::device::device_type>();
80  if(devty == DTy && be == BE) {
81  SelectedDevices.push_back(d);
82  }
83  }
84  if (SelectedDevices.size() > 0) {
85  auto Ctx = context(SelectedDevices);
86  auto d = SelectedDevices[0];
87  queues->emplace_back(Ctx, d);
88  }
89  }
90  }
91  return queues;
92  }

94  static QVec* init_active_queues ()
95  {
96  QVec *active_queues;
97  try {
98  auto def_device { default_selector().select_device() };
99  auto BE = def_device.get_platform().get_backend();
100  auto DevTy = def_device.get_info<info::device::device_type>();
101 
102  // \todo : We need to have a better way to match the default device
103  // to what SYCL returns based on the same scoring logic. Just
104  // storing the first device is not correct when we will have
105  // multiple devices of same type.
106  if(BE == backend::opencl &&
107  DevTy == info::device_type::cpu) {
108  active_queues = new QVec({get_opencl_cpu_queues()[0]});
109  }
110  else if(BE == backend::opencl &&
111  DevTy == info::device_type::gpu) {
112  active_queues = new QVec({get_opencl_gpu_queues()[0]});
113  }
114  else if(BE == backend::level_zero &&
115  DevTy == info::device_type::gpu) {
116  active_queues = new QVec({get_level0_gpu_queues()[0]});
117  }
118  else {
119  active_queues = new QVec();
120  }
121  }
122  catch (runtime_error &re) {
123  // \todo Handle the error
124  active_queues = new QVec();
125  }
126 
127  return active_queues;
128  }

188 DPCTLSyclQueueRef QMgrHelper::getCurrentQueue ()
189 {
190  auto &activated_q = get_active_queues();
191  if(activated_q.empty()) {
192  // \todo handle error
193  std::cerr << "No currently active queues.\n";
194  return nullptr;
195  }
196  auto last = activated_q.size() - 1;
197  return wrap(new queue(activated_q[last]));
198 }

206 __dpctl_give DPCTLSyclQueueRef
207 QMgrHelper::getQueue (DPCTLSyclBackendType BETy,
208  DPCTLSyclDeviceType DeviceTy,
209  size_t DNum)
210 {
211  queue *QRef = nullptr;
212 
213  switch (BETy|DeviceTy)
214  {
215  case DPCTLSyclBackendType::DPCTL_OPENCL | DPCTLSyclDeviceType::DPCTL_CPU:
216  {
217  auto cpuQs = get_opencl_cpu_queues();
218  if (DNum >= cpuQs.size()) {
219  // \todo handle error
220  std::cerr << "OpenCL CPU device " << DNum
221  << " not found on system.\n";
222  return nullptr;
223  }
224  QRef = new queue(cpuQs[DNum]);
225  break;
226  }
227  case DPCTLSyclBackendType::DPCTL_OPENCL | DPCTLSyclDeviceType::DPCTL_GPU:
228  {
229  auto gpuQs = get_opencl_gpu_queues();
230  if (DNum >= gpuQs.size()) {
231  // \todo handle error
232  std::cerr << "OpenCL GPU device " << DNum
233  << " not found on system.\n";
234  return nullptr;
235  }
236  QRef = new queue(gpuQs[DNum]);
237  break;
238  }
239  case DPCTLSyclBackendType::DPCTL_LEVEL_ZERO | DPCTLSyclDeviceType::DPCTL_GPU:
240  {
241  auto l0GpuQs = get_level0_gpu_queues();
242  if (DNum >= l0GpuQs.size()) {
243  // \todo handle error
244  std::cerr << "Level-0 GPU device " << DNum
245  << " not found on system.\n";
246  return nullptr;
247  }
248  QRef = new queue(l0GpuQs[DNum]);
249  break;
250  }
251  default:
252  std::cerr << "Unsupported device type.\n";
253  return nullptr;
254  }
255 
256  return wrap(QRef);
257 }

264 bool QMgrHelper::isCurrentQueue (__dpctl_keep const DPCTLSyclQueueRef QRef)
265 {
266  auto &activated_q = get_active_queues();
267  if(activated_q.empty()) {
268  // \todo handle error
269  std::cerr << "No currently active queues.\n";
270  return false;
271  }
272  auto last = activated_q.size() - 1;
273  auto currQ = activated_q[last];
274  return (*unwrap(QRef) == currQ);

348 __dpctl_give DPCTLSyclQueueRef
349 QMgrHelper::pushSyclQueue (DPCTLSyclBackendType BETy,
350  DPCTLSyclDeviceType DeviceTy,
351  size_t DNum)
352 {
353  queue *QRef = nullptr;
354  auto &activeQ = get_active_queues();
355  if(activeQ.empty()) {
356  std::cerr << "Why is there no previous global context?\n";
357  return nullptr;
358  }
359 
360  switch (BETy|DeviceTy)
361  {
362  case DPCTLSyclBackendType::DPCTL_OPENCL | DPCTLSyclDeviceType::DPCTL_CPU:
363  {
364  if (DNum >= get_opencl_cpu_queues().size()) {
365  // \todo handle error
366  std::cerr << "OpenCL CPU device " << DNum
367  << " not found on system\n.";
368  return nullptr;
369  }
370  activeQ.emplace_back(get_opencl_cpu_queues()[DNum]);
371  QRef = new queue(activeQ[activeQ.size()-1]);
372  break;
373  }
374  case DPCTLSyclBackendType::DPCTL_OPENCL | DPCTLSyclDeviceType::DPCTL_GPU:
375  {
376  if (DNum >= get_opencl_gpu_queues().size()) {
377  // \todo handle error
378  std::cerr << "OpenCL GPU device " << DNum
379  << " not found on system\n.";
380  return nullptr;
381  }
382  activeQ.emplace_back(get_opencl_gpu_queues()[DNum]);
383  QRef = new queue(activeQ[get_active_queues().size()-1]);
384  break;
385  }
386  case DPCTLSyclBackendType::DPCTL_LEVEL_ZERO | DPCTLSyclDeviceType::DPCTL_GPU:
387  {
388  if (DNum >= get_level0_gpu_queues().size()) {
389  // \todo handle error
390  std::cerr << "Level-0 GPU device " << DNum
391  << " not found on system\n.";
392  return nullptr;
393  }
394  activeQ.emplace_back(get_level0_gpu_queues()[DNum]);
395  QRef = new queue(activeQ[get_active_queues().size()-1]);
396  break;
397  }
398  default:
399  {
400  std::cerr << "Unsupported device type.\n";
401  return nullptr;
402  }
403  }
404 
405  return wrap(QRef);
406 }

417 QMgrHelper::popSyclQueue ()
418 {
419  // The first queue which is the "default" queue can not be removed.
420  if(get_active_queues().size() <= 1 ) {
421  std::cerr << "No active contexts.\n";
