41 get_set_of_non_hostbackends ()
42 {
43  std::set<DPCTLSyclBackendType> be_set;
44  for (auto p : platform::get_platforms()) {
45  if(p.is_host())
46  continue;
47  auto be = p.get_backend();
48  switch (be)
49  {
50  case backend::host:
51  break;
52  case backend::cuda:
53  be_set.insert(DPCTLSyclBackendType::DPCTL_CUDA);
54  break;
55  case backend::level_zero:
56  be_set.insert(DPCTLSyclBackendType::DPCTL_LEVEL_ZERO);
57  break;
58  case backend::opencl:
59  be_set.insert(DPCTLSyclBackendType::DPCTL_OPENCL);
60  break;
61  default:
62  break;
63  }
64  }
65  return be_set;
66 }
