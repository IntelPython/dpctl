Overview

enum DPCTLKernelArgType
{
    DPCTL_CHAR,
    DPCTL_SIGNED_CHAR,
    DPCTL_UNSIGNED_CHAR,
    DPCTL_SHORT,
    DPCTL_INT,
    DPCTL_UNSIGNED_INT,
    DPCTL_UNSIGNED_INT8,
    DPCTL_LONG,
    DPCTL_UNSIGNED_LONG,
    DPCTL_LONG_LONG,
    DPCTL_UNSIGNED_LONG_LONG,
    DPCTL_SIZE_T,
    DPCTL_FLOAT,
    DPCTL_DOUBLE,
    DPCTL_LONG_DOUBLE,
