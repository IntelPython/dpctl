enum DPCTLSyclDeviceType¶

DPCTL device types that are equivalent to Sycl’s device_type.

#include <dpctl_sycl_enum_types.h>

enum DPCTLSyclDeviceType
{
    DPCTL_CPU         = 1<<0,
    DPCTL_GPU         = 1<<1,
    DPCTL_ACCELERATOR = 1<<2,
    DPCTL_CUSTOM      = 1<<3,
    DPCTL_AUTOMATIC   = 1<<4,
    DPCTL_HOST_DEVICE = 1<<5,
    DPCTL_ALL         = 1<<6,
};
